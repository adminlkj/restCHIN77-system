   197→          </p>
   198→        </CardContent>
   199→      </Card>
   200→
   201→      {loading ? (
   202→        <div className="grid grid-cols-3 gap-4">{Array.from({ length: 6 }).map((_, i) => <Card key={i}><CardContent className="p-5"><div className="h-16 bg-muted animate-pulse rounded" /></CardContent></Card>)}</div>
   203→      ) : (
   204→        <>
   205→          {/* Income Statement */}
   206→          {activeReport === 'income' && (
   207→            <div className="space-y-4">
   208→              {/* صندوق توضيحي */}
   209→              <div className="rounded-lg border border-emerald-200 bg-emerald-50/60 p-3 text-xs text-emerald-800 flex items-start gap-2">
   210→                <Info className="size-4 shrink-0 mt-0.5" />
   211→                <div>

   213→                  <p>{t('الإيرادات = دائن في حسابات الإيراد (4xxx). التكاليف = مدين في حسابات المصروفات (5xxx). كل رقم من قيود اليومية المرحّلة (isPosted=true).', 'Revenue = credit in revenue accounts (4xxx).
