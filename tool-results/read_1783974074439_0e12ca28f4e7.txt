   123→// فرز نتائج حسب الحقل: -field = تنازلي، field = تصاعدي
   124→function sortBy(items, sort) {
   125→  if (!sort) return items;
   126→  const desc = sort.startsWith('-');
   127→  const field = desc ? sort.slice(1) : sort;
   128→  return [...items].sort((a, b) => {
   129→    let av = a[field], bv = b[field];
   130→    if (av == null) av = '';
   131→    if (bv == null) bv = '';
   132→    if (typeof av === 'number' && typeof bv === 'number') {
   133→      return desc ? bv - av : av - bv;
   134→    }
   135→    av = String(av); bv = String(bv);
   136→    return desc ? bv.localeCompare(av) : av.localeCompare(bv);
   137→  });
   138→}

   140→// مطابقة سجل بالاستعلام (query) — يدعم المساواة البسيطة
   141→function matchesQuery(item, query) {
   142→  if (!query || typeof query !== 'object') return true;
   143→  for (const [key, value] of Object.entries(query)) {
   144→    if (item[key] !== value) return false;
   145→  }
   146→  return true;
   147→}

   244→function entityUpdateMany(entityName, query, update) {
   245→  const db = loadDB();
   246→  const items = getCollection(db, entityName);
   247→  let count = 0;
   248→  for (let i = 0; i < items.length; i++) {
   249→    if (matchesQuery(items[i], query)) {
   250→      items[i] = { ...items[i], ...update, updated_date: new Date().toISOString() };
   251→      count++;
   252→    }
   253→  }
   254→  saveDB(db);
   255→  return { modifiedCount: count };
   256→}

   272→function entityDeleteMany(entityName, query) {
   273→  const db = loadDB();
   274→  let items = getCollection(db, entityName);
   275→  const before = items.length;
   276→  items = items.filter(i => !matchesQuery(i, query));
   277→  const after = items.length;
   278→  db.entities[entityName] = items;
   279→  saveDB(db);
   280→  return { deletedCount: before - after };

   381→// ─── تحميل ملف (Upload) ────────────────────────────────────────────────────
   382→async function uploadFile({ file } = {}) {
   383→  if (!file) {
   384→    const err = new Error('No file provided');
   385→    err.status = 400;
   386→    throw err;
   387→  }
   388→  // حوّل الملف إلى data URL محلي
   389→  return new Promise((resolve, reject) => {
   390→    const reader = new FileReader();
   391→    reader.onload = () => resolve({ file_url: reader.result });
   392→    reader.onerror = () => reject(new Error('فشل قراءة الملف'));
   393→    reader.readAsDataURL(file);
   394→  });
   395→}
